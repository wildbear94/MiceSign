useEffect(() => {
  if (!calculationRules || calculationRules.length === 0) return;

  const results = executeCalculations(calculationRules, formValues);

  // 변경 감지: 실제로 달라진 값이 있을 때만 setState

    if (formValues[fid] !== v) {
      changed = true;
      break;
    }

  if (changed) {
    setFormValues(prev => ({ ...prev, ...results }));
