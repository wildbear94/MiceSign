    ```tsx
    import type { Control, UseFormRegister } from 'react-hook-form';
    import type { FieldDefinition } from '../../types/dynamicForm';
    import DynamicTableField from './DynamicTableField'; // Task 2에서 구현 — import만 먼저 작성
    // Task 2 전에는 타입 에러가 날 수 있으므로 import를 주석 처리하거나 stub을 먼저 둔다.

    export interface DynamicFieldRendererProps {
      field: FieldDefinition;
      mode: 'edit' | 'readOnly';
      register?: UseFormRegister<Record<string, unknown>>;
      control?: Control<Record<string, unknown>>;
      error?: string;
      dynamicRequired?: boolean;
      disabled?: boolean;
      value?: unknown;
    }

    export default function DynamicFieldRenderer(props: DynamicFieldRendererProps) {
      const { field, mode } = props;
      const isRequired = props.dynamicRequired ?? field.required ?? false;

      // hidden: 실제 렌더 없음 (값만 form에 유지)
      if (field.type === 'hidden') return null;

      // staticText: 읽기 전용 설명
      if (field.type === 'staticText') {
        return <div className="text-sm text-gray-600 py-2">{field.config?.text ?? field.label}</div>;
      }

      // table: 별도 컴포넌트 위임
      if (field.type === 'table') {
        if (mode === 'edit' && props.control) {
          return <DynamicTableField field={field} control={props.control} disabled={props.disabled} />;
        }
        // readOnly table은 Plan 03에서 처리. 여기서는 edit만.
        return <div data-testid={`dyn-table-${field.id}`}>[table readOnly handled by DynamicCustomReadOnly]</div>;
      }

      // readOnly 모드 공통 렌더링 (라벨 + 값 텍스트)
      if (mode === 'readOnly') {
        const displayValue = formatReadOnlyValue(field, props.value);
        return (
          <div className="py-2">
            <div className="text-sm font-medium text-gray-700">{field.label}</div>
            <div className="text-base text-gray-900 whitespace-pre-wrap">{displayValue || '-'}</div>
          </div>
        );
      }

      // edit 모드 — register 필수
      if (!props.register) {
        throw new Error(`DynamicFieldRenderer: register is required in edit mode (field=${field.id})`);
      }

      const commonProps = {
        id: field.id,
        disabled: props.disabled,
        'aria-required': isRequired,
      };

      const labelBlock = (
        <label htmlFor={field.id} className="block text-sm font-medium text-gray-700 mb-1">
          {field.label}{isRequired && <span className="text-red-500 ml-1">*</span>}
        </label>
      );

      const errorBlock = props.error ? (
        <p className="mt-1 text-sm text-red-600">{props.error}</p>
      ) : null;

      switch (field.type) {
        case 'text':
          return (
            <div className="mb-4">
              {labelBlock}
              <input type="text" {...commonProps}
                placeholder={field.config?.placeholder ?? ''}
                {...props.register(field.id)}
                className="w-full border rounded px-3 py-2" />
              {errorBlock}
            </div>
          );
        case 'textarea':
          return (
            <div className="mb-4">
              {labelBlock}
              <textarea {...commonProps}
                placeholder={field.config?.placeholder ?? ''}
                rows={field.config?.rows ?? 4}
                {...props.register(field.id)}
                className="w-full border rounded px-3 py-2" />
              {errorBlock}
            </div>
          );
        case 'number':
          return (
            <div className="mb-4">
              {labelBlock}
              <input type="number" {...commonProps}
                placeholder={field.config?.placeholder ?? ''}
                {...props.register(field.id, { valueAsNumber: true })}
                className="w-full border rounded px-3 py-2" />
              {errorBlock}
            </div>
          );
        case 'date':
          return (
            <div className="mb-4">
              {labelBlock}
              <input type="date" {...commonProps}
                {...props.register(field.id)}
                className="w-full border rounded px-3 py-2" />
              {errorBlock}
            </div>
          );
        case 'select':
          return (
            <div className="mb-4">
              {labelBlock}
              <select {...commonProps}
                {...props.register(field.id)}
                className="w-full border rounded px-3 py-2">
                <option value="">선택하세요</option>
                {(field.options ?? []).map((opt) => (
                  <option key={opt.value} value={opt.value}>{opt.label}</option>
                ))}
              </select>
              {errorBlock}
            </div>
          );
        default:
          return <div className="text-red-500">Unsupported field type: {field.type}</div>;
      }
    }

    function formatReadOnlyValue(field: FieldDefinition, value: unknown): string {
      if (value === undefined || value === null || value === '') return '';
      if (field.type === 'select') {
        const opt = (field.options ?? []).find(o => o.value === value);
        return opt?.label ?? String(value);
      }
      if (field.type === 'date' && typeof value === 'string') return value;
      return String(value);
    }
    ```

    ```tsx
    import { useFieldArray, useFormContext, type Control } from 'react-hook-form';
    import type { FieldDefinition, ColumnDefinition } from '../../types/dynamicForm';
    import { buildDefaultRow } from '../../utils/schemaToZod';

    interface DynamicTableFieldProps {
      field: FieldDefinition;
      control: Control<Record<string, unknown>>;
      disabled?: boolean;
    }

    export default function DynamicTableField({ field, control, disabled }: DynamicTableFieldProps) {
      const columns = field.columns ?? [];
      const { fields, append, remove } = useFieldArray({
        control: control as any,
        name: field.id,
      });
      const { register } = useFormContext();

      return (
        <div className="mb-4">
          <div className="flex items-center justify-between mb-2">
            <label className="block text-sm font-medium text-gray-700">
              {field.label}{field.required && <span className="text-red-500 ml-1">*</span>}
            </label>
            {!disabled && (
              <button
                type="button"
                onClick={() => append(buildDefaultRow(columns))}
                className="px-3 py-1 text-sm bg-blue-500 text-white rounded hover:bg-blue-600"
              >
                + 행 추가
              </button>
            )}
          </div>
          <table className="w-full border-collapse border border-gray-300">
            <thead className="bg-gray-50">
              <tr>
                {columns.map((col) => (
                  <th key={col.id} className="border border-gray-300 px-3 py-2 text-left text-sm font-medium">
                    {col.label}{col.required && <span className="text-red-500 ml-1">*</span>}
                  </th>
                ))}
                {!disabled && <th className="border border-gray-300 px-3 py-2 w-16"></th>}
              </tr>
            </thead>
            <tbody>
              {fields.length === 0 && (
                <tr>
                  <td colSpan={columns.length + 1} className="border border-gray-300 px-3 py-4 text-center text-gray-500 text-sm">
                    행이 없습니다. "행 추가" 버튼을 눌러 추가하세요.
                  </td>
                </tr>
              )}
              {fields.map((row, rowIdx) => (
                <tr key={row.id}>
                  {columns.map((col) => (
                    <td key={col.id} className="border border-gray-300 px-2 py-1">
                      {renderCell(col, `${field.id}.${rowIdx}.${col.id}`, register, disabled)}
                    </td>
                  ))}
                  {!disabled && (
                    <td className="border border-gray-300 px-2 py-1 text-center">
                      <button
                        type="button"
                        onClick={() => remove(rowIdx)}
                        className="text-red-600 hover:text-red-800 text-sm"
                        aria-label={`행 ${rowIdx + 1} 삭제`}
                      >
                        삭제
                      </button>
                    </td>
                  )}
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      );
    }

    function renderCell(col: ColumnDefinition, name: string, register: any, disabled?: boolean) {
      const commonCls = 'w-full border-0 px-2 py-1 text-sm focus:outline-none focus:ring-1 focus:ring-blue-500';
      switch (col.type) {
        case 'number':
          return <input type="number" disabled={disabled} className={commonCls}
            placeholder={col.config?.placeholder ?? ''}
            {...register(name, { valueAsNumber: true })} />;
        case 'date':
          return <input type="date" disabled={disabled} className={commonCls} {...register(name)} />;
        case 'select':
          return (
            <select disabled={disabled} className={commonCls} {...register(name)}>
              <option value="">선택</option>
              {(col.options ?? []).map((opt) => (
                <option key={opt.value} value={opt.value}>{opt.label}</option>
              ))}
            </select>
          );
        case 'text':
        default:
          return <input type="text" disabled={disabled} className={commonCls}
            placeholder={col.config?.placeholder ?? ''}
            {...register(name)} />;
      }
    }
    ```

    ```ts
    import type { SchemaField } from '../../../admin/components/SchemaFieldEditor/types';
    import type { FieldDefinition, ColumnDefinition, OptionItem } from '../../types/dynamicForm';

    /**
     * admin SchemaFieldEditor에서 사용하는 SchemaField 타입을
     * user document 렌더러에서 사용하는 FieldDefinition 타입으로 변환.
     * 주된 차이:
     *  - config가 non-optional → optional (값 그대로 전달)
     *  - options: {value,label}[] → OptionItem[] (sortOrder 없음 그대로, 타입만 호환)
     *  - columns[].config: non-optional → optional
     *  - columns[].required: non-optional → optional
     *  - columns[].type: TableColumnType(checkbox 포함) → FieldType (checkbox 제외; 현재 admin도 checkbox 미사용 가정)
     */
    export function adaptSchemaFieldToFieldDefinition(f: SchemaField): FieldDefinition {
      return {
        id: f.id,
        type: f.type as FieldDefinition['type'],
        label: f.label,
        required: f.required,
        config: f.config ?? {},
        options: (f.options ?? []).map((o, idx): OptionItem => ({
          value: o.value,
          label: o.label,
          sortOrder: idx,
        })),
        columns: (f.columns ?? []).map((c): ColumnDefinition => ({
          id: c.id,
          type: c.type as ColumnDefinition['type'],
          label: c.label,
          required: c.required,
          config: c.config ?? {},
          options: (c.options ?? []).map((o, idx): OptionItem => ({
            value: o.value,
            label: o.label,
            sortOrder: idx,
          })),
        })),
      };
    }

    export function adaptSchemaFields(fields: SchemaField[]): FieldDefinition[] {
      return fields.map(adaptSchemaFieldToFieldDefinition);
    }
    ```

    **주의:** admin `SchemaField`와 user `FieldDefinition` 필드 구조가 완벽히 일치하지 않을 수 있다. Task 실행 시 admin types.ts를 먼저 읽고 누락 필드는 무시하거나 default 값으로 채운다. 컴파일 에러 발생 시 최소 필드(id, type, label, required)만 매핑하고 나머지는 `as any` 주석과 함께 처리 (하지만 가급적 타입 안전 유지).

    타입이 실제로 호환되지 않아 `as FieldDefinition['type']` 캐스팅이 필요한 경우 주석으로 이유 명시: `// admin TableColumnType['checkbox']는 현재 사용되지 않음`.
