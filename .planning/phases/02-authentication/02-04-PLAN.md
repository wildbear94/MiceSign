    **1. Create password validation schema** (`schemas/passwordSchema.ts`):
    ```typescript
    import { z } from 'zod';

    // D-27: min 8 chars, at least 2 of: uppercase, lowercase, number, special
    export const passwordStrengthCheck = (password: string) => {
      const categories = [
        /[A-Z]/.test(password),      // uppercase
        /[a-z]/.test(password),      // lowercase
        /[0-9]/.test(password),      // number
        /[!@#$%^&*()_+\-=\[\]{}|;:'",.<>?/~`]/.test(password), // special
      ].filter(Boolean).length;
      return password.length >= 8 && categories >= 2;
    };

    export const passwordChangeSchema = z.object({
      currentPassword: z.string().min(1, 'passwordChange.wrongCurrent'),
      newPassword: z.string().min(8, 'requirements.minLength').refine(passwordStrengthCheck, 'passwordChange.ruleNotMet'),
      confirmPassword: z.string().min(1, 'validation.passwordRequired'),
    }).refine((data) => data.newPassword === data.confirmPassword, {
      message: 'passwordChange.mismatch',
      path: ['confirmPassword'],
    });
    ```
