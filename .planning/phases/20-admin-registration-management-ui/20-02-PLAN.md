```typescript
  const result: { id: number; name: string }[] = [];

}
```

2. **Update `RegistrationService.getRegistrations()`** -- replace the simple `page.map(registrationMapper::toListResponse)` with logic that populates assignment fields for APPROVED registrations:

```java
public Page<RegistrationListResponse> getRegistrations(RegistrationStatus status, Pageable pageable) {
    Page<RegistrationRequest> page;
    if (status == null) {
        page = registrationRequestRepository.findAll(pageable);
    } else {
        page = registrationRequestRepository.findByStatus(status, pageable);
    }
    return page.map(this::toListResponseWithAssignment);
}

private RegistrationListResponse toListResponseWithAssignment(RegistrationRequest reg) {
    String employeeNo = null;
    String departmentName = null;
    String positionName = null;

    if (reg.getStatus() == RegistrationStatus.APPROVED) {
        User user = userRepository.findByEmail(reg.getEmail()).orElse(null);
        if (user != null) {
            employeeNo = user.getEmployeeNo();
            if (user.getDepartmentId() != null) {
                departmentName = departmentRepository.findById(user.getDepartmentId())
                    .map(d -> d.getName())
                    .orElse(null);
            }
            if (user.getPositionId() != null) {
                positionName = positionRepository.findById(user.getPositionId())
                    .map(p -> p.getName())
                    .orElse(null);
            }
        }
    }

    return new RegistrationListResponse(
        reg.getId(),
        reg.getName(),
        reg.getEmail(),
        reg.getStatus(),
        reg.getRejectionReason(),
        reg.getCreatedAt(),
        reg.getProcessedAt(),
        employeeNo,
        departmentName,
        positionName
    );
}
```

Note: The `RegistrationMapper.toListResponse()` is no longer used for this method. Keep it in the mapper interface for potential other uses, but the service now handles mapping directly with the assignment lookup. If there are compiler warnings about unused methods, that is acceptable.

The N+1 query concern is minimal here because: (a) the admin list is paginated (max 20 items per page), (b) only APPROVED items trigger the extra lookup, (c) this is an admin-only page with low traffic.

```typescript
async function handleApprove() {
  if (!validateApproveForm()) return;
  try {
    await approveMutation.mutateAsync({
      id: registration.id,
      data: { employeeNo, departmentId: Number(departmentId), positionId: Number(positionId) },
    });
    toast.success(t('registration.toast.approved'));
    onClose();
  } catch (err) {
    const axiosError = err as AxiosError<ApiResponse<null>>;
    const errorCode = axiosError.response?.data?.error?.code;
    if (errorCode === 'REGISTRATION_ALREADY_PROCESSED' || axiosError.response?.status === 409) {
      toast.error(t('registration.toast.alreadyProcessed'));
    } else {
      toast.error(t('registration.toast.error'));
    }
  }
}
```
