Drag-and-drop implementation (following PositionTable.tsx pattern):
```typescript
import { DragDropContext, Droppable, Draggable, type DropResult } from '@hello-pangea/dnd';

function handleDragEnd(result: DropResult) {
  if (!result.destination) return;
  const sequentialItems = items.filter(i => i.lineType !== 'REFERENCE');
  const referenceItems = items.filter(i => i.lineType === 'REFERENCE');
  const reordered = Array.from(sequentialItems);
  const [removed] = reordered.splice(result.source.index, 1);
  reordered.splice(result.destination.index, 0, removed);
  onChange([...reordered, ...referenceItems]);
}
```

**Conversion utility** — Add a helper function `toApprovalLineRequests`:
```typescript
export function toApprovalLineRequests(items: ApprovalLineItem[]): ApprovalLineRequest[] {
  let step = 1;
  return items.map(item => ({
    approverId: item.userId,
    lineType: item.lineType,
    stepOrder: item.lineType === 'REFERENCE' ? 0 : step++,
  }));
}
```

**Mapping utility** — Add a helper to convert `ApprovalLineResponse[]` back to `ApprovalLineItem[]` for editor initialization:

export function toApprovalLineItems(responses: ApprovalLineResponse[]): ApprovalLineItem[] {

    userName: r.approver.name,
    departmentName: r.approver.departmentName,
    positionName: r.approver.positionName,
    lineType: r.lineType,
  }));
}
