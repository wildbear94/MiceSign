```typescript
  },
  withCredentials: true,
  paramsSerializer: {

        if (value === null || value === undefined) return;
        if (Array.isArray(value)) {
          value.forEach((v) => {
            if (v !== null && v !== undefined && v !== '') {

          sp.append(key, String(value));
        }
      });
      return sp.toString();
    },
  },
});
```

```typescript
    ...(statuses && statuses.length > 0 ? { status: statuses } : {}),
    ...(serverTab ? { tab: serverTab } : {}),
  };
```

```tsx
  placeholder?: string;
  className?: string;

export function DrafterCombo({ value, onChange, placeholder = '기안자 이름', className }: DrafterComboProps) {
  const [query, setQuery] = useState('');
  const [candidates, setCandidates] = useState<UserSearchItem[]>([]);
  const [displayName, setDisplayName] = useState('');
  const [isOpen, setIsOpen] = useState(false);
  const debounceRef = useRef<ReturnType<typeof setTimeout>>();

  // When URL value changes externally and we don't have display name yet,
  // fetch the user's name (for link sharing - new session loads drafterId from URL).
  // Revision 1 WARNING 5: displayName 을 deps 에 넣으면 setDisplayName → re-run loop 발생.
  // 프로젝트 eslint 가 `react-hooks/exhaustive-deps` 를 error 로 설정했을 수 있으므로
  // disable 주석을 명시 (대안: displayName 을 useRef 로 두면 deps 에서 제외 가능하나
  // 렌더에서 사용하므로 useState 유지 + 의도적 제외가 더 명확).
  useEffect(() => {
    if (value && !displayName) {
      userSearchApi.search(value.toString(), 1).then(r => {
        const users = r.data.data;
        const match = users.find(u => String(u.id) === value);
        if (match) setDisplayName(match.name);
      }).catch(() => { /* silent — user will see id */ });
    }
    if (!value) setDisplayName('');
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [value]);

  // Debounced search
  useEffect(() => {
    clearTimeout(debounceRef.current);
    if (!query || query.length < 2) {
      setCandidates([]);
      return;
    }
    debounceRef.current = setTimeout(() => {
      userSearchApi.search(query, 20).then(r => {
        setCandidates(r.data.data);
        setIsOpen(true);
      }).catch(() => setCandidates([]));
    }, 300);
    return () => clearTimeout(debounceRef.current);
  }, [query]);

  const handleSelect = (user: UserSearchItem) => {
    onChange(user.id, user.name);
    setDisplayName(user.name);
    setQuery('');
    setCandidates([]);
    setIsOpen(false);
  };

  const handleClear = () => {
    onChange(null);
    setDisplayName('');
    setQuery('');
    setCandidates([]);
    setIsOpen(false);
  };

  return (
    <div className={`relative ${className ?? ''}`}>
      <div className="flex items-center border border-gray-300 rounded-md h-9 px-3">
        <input
          value={value ? displayName : query}
          onChange={e => {
            if (value) onChange(null);  // clear selection on new input
            setQuery(e.target.value);
            setIsOpen(true);
          }}
          onFocus={() => { if (candidates.length > 0) setIsOpen(true); }}
          onBlur={() => setTimeout(() => setIsOpen(false), 150)}  // delay for click
          placeholder={placeholder}
          className="flex-1 outline-none text-sm"
        />

                  className="ml-1 text-gray-400 hover:text-gray-600">
            <X size={14} />
          </button>
        )}

          {candidates.map(u => (

              {u.departmentName && <span className="text-gray-500 ml-2">({u.departmentName})</span>}
            </li>
          ))}
        </ul>
      )}
    </div>
  );
```

```tsx
import { useSearchParams } from 'react-router';
import { useEffect, useMemo, useRef, useState } from 'react';
import type { DocumentStatus } from '../types/document';
import { DrafterCombo } from '../components/DrafterCombo';
import { StatusFilterPills } from '../components/StatusFilterPills';
// ... 기존 imports 유지

export default function DocumentListPage() {
  const [searchParams, setSearchParams] = useSearchParams();

  // Derive filters from URL (single SoT)
  const filters = useMemo(() => ({
    tab: (searchParams.get('tab') ?? 'my') as 'my' | 'search',
    keyword: searchParams.get('keyword') ?? '',
    statuses: searchParams.getAll('status') as DocumentStatus[],  // Pitfall 5: getAll
    templateCode: searchParams.get('templateCode') ?? '',
    dateFrom: searchParams.get('dateFrom') ?? '',
    dateTo: searchParams.get('dateTo') ?? '',
    drafterId: searchParams.get('drafterId'),  // string | null
    page: Number(searchParams.get('page') ?? '0'),
  }), [searchParams]);

  // Local input state for keyword (debounce source only)
  const [keywordInput, setKeywordInput] = useState(filters.keyword);
  const debounceRef = useRef<ReturnType<typeof setTimeout>>();

  // Sync local state when URL keyword changes externally (back/forward)
  useEffect(() => {
    setKeywordInput(filters.keyword);
  }, [filters.keyword]);

  // Debounced keyword → URL
  useEffect(() => {
    clearTimeout(debounceRef.current);
    debounceRef.current = setTimeout(() => {
      if (keywordInput === filters.keyword) return;
      setSearchParams(prev => {                  // Pitfall 4: callback pattern
        const next = new URLSearchParams(prev);
        if (keywordInput) next.set('keyword', keywordInput);
        else next.delete('keyword');
        next.set('page', '0');                   // D-C8: reset page on filter change
        return next;
      }, { replace: true });                     // D-C3: history 오염 방지
    }, 300);
    return () => clearTimeout(debounceRef.current);
  }, [keywordInput, filters.keyword, setSearchParams]);

  // Helper for non-keyword filter changes (push history — 뒤로가기 가능)
  const updateFilter = (updates: Record<string, string | string[] | null>) => {
    setSearchParams(prev => {
      const next = new URLSearchParams(prev);
      Object.entries(updates).forEach(([key, value]) => {
        next.delete(key);
        if (Array.isArray(value)) {
          value.forEach(v => {
            if (v !== null && v !== '') next.append(key, v);
          });
        } else if (value !== null && value !== '') {
          next.set(key, value);
        }
      });
      next.set('page', '0');  // D-C8
      return next;
    });
  };

  // Page change separately (doesn't reset page to 0)
  const changePage = (newPage: number) => {
    setSearchParams(prev => {
      const next = new URLSearchParams(prev);
      if (newPage === 0) next.delete('page');
      else next.set('page', String(newPage));
      return next;
    });
  };

  // Backend tab mapping (D-C5): frontend 'search' → backend 'all'
  const searchParams_forQuery = useMemo(() => {
    const params: DocumentSearchParams = {
      page: filters.page,
      size: 20,
      tab: filters.tab,
    };
    if (filters.keyword) params.keyword = filters.keyword;
    if (filters.statuses.length > 0) params.statuses = filters.statuses;
    if (filters.templateCode) params.templateCode = filters.templateCode;
    if (filters.dateFrom) params.dateFrom = filters.dateFrom;
    if (filters.dateTo) params.dateTo = filters.dateTo;
    if (filters.drafterId) params.drafterId = Number(filters.drafterId);
    return params;
  }, [filters]);

  const isSearchActive = filters.tab === 'search';
  // 기존 useMyDocuments + useSearchDocuments 훅 분기 (D-C6 두 훅 분리 유지)
  const myDocsQuery = useMyDocuments({
    page: filters.page, size: 20,
    status: filters.statuses[0]  // tab=my 는 단일 status 지원만 — 복수는 search 탭에서
  }, !isSearchActive);
  const searchQuery = useSearchDocuments(searchParams_forQuery, isSearchActive);

  const activeQuery = isSearchActive ? searchQuery : myDocsQuery;
  // ... 나머지 rendering 기존 로직 유지

  return (
    <div>
      {/* 탭 선택 */}
      <Tabs value={filters.tab} onChange={(v) => updateFilter({ tab: v === 'my' ? null : v })} />
      {/* keyword 입력 (debounce 로 URL 반영) */}
      <input
        value={keywordInput}
        onChange={(e) => setKeywordInput(e.target.value)}
        placeholder="제목/문서번호/기안자 검색"
      />
      {/* 복수 상태 pills (search 탭 한정) */}
      {isSearchActive && (
        <StatusFilterPills
          value={filters.statuses}
          onChange={(statuses) => updateFilter({ status: statuses })}
        />
      )}
      {/* 양식 select */}
      <select value={filters.templateCode} onChange={(e) => updateFilter({ templateCode: e.target.value || null })}>
        {TEMPLATE_OPTIONS.map(...)}
      </select>
      {/* 기간 */}
      <input type="date" value={filters.dateFrom} onChange={(e) => updateFilter({ dateFrom: e.target.value || null })} />
      <input type="date" value={filters.dateTo} onChange={(e) => updateFilter({ dateTo: e.target.value || null })} />
      {/* 기안자 콤보 (search 탭 한정) */}
      {isSearchActive && (
        <DrafterCombo
          value={filters.drafterId}
          onChange={(id) => updateFilter({ drafterId: id ? String(id) : null })}
        />
      )}
      {/* 결과 + Pagination */}
      <Pagination current={filters.page} totalElements={activeQuery.data?.totalElements ?? 0} pageSize={20} onChange={changePage} />
    </div>
  );
}
```
